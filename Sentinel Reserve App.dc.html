public fun seal_approve(p: &Mandate, t: Trade) {
  assert!(t.amount <= p.daily_left, E_BUDGET);
  assert!(p.assets.has(t.class), E_ASSET);
  assert!(p.markets.has(t.market), E_MARKET);
  assert!(clock.now < p.expiry, E_EXPIRED);
  assert!(!p.used.has(t.nonce), E_NONCE_REPLAY);
  check_rotate_transfer(p, t);
}
